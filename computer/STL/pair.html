# 二元组pair

#include <utility>

顾名思义，是存储二元组的

# 1. 常用方法

# 构造

pair<第一个值类型, 第二个值类型> pr

pair<int, int> p1;
pair<char, int> p2;
...
//三元组，套娃
pair<pair<int, int>, int> p3;

# 赋值

老式

pair<int, char> pr = make_pair(1, 'a');

C++ 11

pair<int, char> pr = {1, 'a'};

# 取值

直接取值

  • 取第一个值：.first
  • 取第二个值：.second
pair<int, char> pr = {1, 'a'};
int a = pr.first;
char c = pr.second;
